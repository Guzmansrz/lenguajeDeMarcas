Funcion resta(entero num1, entero num2) : entero

Funcion multiplicacion(entero num1, entero num2) : entero
	retornar num1*num2

Funcion division(entero num1, entero num2) : real
	retornar num1/num2

Funcion cuadrado(entero num1) : entero
	retornar num1^2

Funcion potencia(entero num1, entero num2) : entero
